const EMOJIS = ['🔗','📊','📋','🗂','📁','🖥','⚙','🔧','🔍','📝','📌','🚀','💡','🌐','📬','🗓','📦','🎯','🧮','🛠','📐','🔐','📈','🧩','🎨','📷','🎬','🎵','📚','🗒']; const CATEGORIES = ['Công cụ','Tài liệu','Hệ thống','Tra cứu','Quản lý','Khác']; const COLORS = [ ['#63b3ed','#76e4c4'],['#a78bfa','#f472b6'],['#fb923c','#fbbf24'], ['#34d399','#22d3ee'],['#f87171','#fb923c'],['#818cf8','#38bdf8'] ]; let cards = JSON.parse(localStorage.getItem('tienich_cards') || '[]'); let syncKey = localStorage.getItem('tienich_synckey') || ''; let pickerTarget = null; let saveDebounceTimeout = null; function setStatus(type, text) { const dot = document.getElementById('statusDot'); const txt = document.getElementById('syncStatus'); if(!dot || !txt) return; dot.className = 'status-dot'; if(type === 'online') dot.classList.add('status-online'); if(type === 'loading') dot.classList.add('status-loading'); txt.textContent = text; } // Hàm lưu dữ liệu function save() { localStorage.setItem('tienich_cards', JSON.stringify(cards)); updateCount(); if (!syncKey) { setStatus('', 'Đang lưu Offline (Nhập mã phía trên hoặc bấm Tải về để tạo mã mới).'); return; } setStatus('loading', 'Đang đồng bộ lên Cloud...'); clearTimeout(saveDebounceTimeout); saveDebounceTimeout = setTimeout(() => { // API npoint sử dụng phương thức POST/PUT để ghi đè dữ liệu JSON nguyên bản fetch(`https://api.npoint.io/${syncKey}`, { method: 'POST', body: JSON.stringify({ data: cards }), headers: { 'Content-Type': 'application/json' } }) .then(res => { if(res.ok) setStatus('online', 'Đã đồng bộ lên Cloud thành công!'); else setStatus('', 'Lỗi đồng bộ Cloud (Thử lại sau), đã lưu tạm ở máy.'); }) .catch(() => setStatus('', 'Không có kết nối mạng. Đã lưu tạm ở máy.')); }, 1000); } // Hàm tải dữ liệu / Tạo thùng chứa mới function loadFromCloud() { setStatus('loading', 'Đang kết nối đám mây...'); // NẾU CHƯA CÓ KEY -> Tự động đăng ký một thùng chứa (JSON bin) hoàn toàn mới trên npoint if (!syncKey) { fetch('https://api.npoint.io', { method: 'POST', body: JSON.stringify({ data: cards }), headers: { 'Content-Type': 'application/json' } }) .then(res => res.json()) .then(result => { if(result && result.id) { syncKey = result.id; document.getElementById('syncKeyInput').value = syncKey; localStorage.setItem('tienich_synckey', syncKey); setStatus('online', `Đã cấp mã mới: ${syncKey}. Hãy lưu mã này sang máy khác!`); } }) .catch(() => setStatus('', 'Lỗi khởi tạo Cloud. Đang chạy Offline.')); return; } // NẾU ĐÃ CÓ KEY -> Tiến hành tải dữ liệu về fetch(`https://api.npoint.io/${syncKey}`) .then(res => { if (!res.ok) throw new Error('Mã không tồn tại'); return res.json(); }) .then(cloudObj => { if (cloudObj && Array.isArray(cloudObj.data)) { cards = cloudObj.data; localStorage.setItem('tienich_cards', JSON.stringify(cards)); renderCards(); setStatus('online', 'Đã cập nhật dữ liệu mới nhất từ Cloud!'); } else { // Nếu thùng chứa rỗng hoặc lỗi định dạng, đẩy data hiện tại lên luôn save(); } }) .catch(err => { console.error(err); setStatus('', 'Mã Cloud không hợp lệ hoặc đã bị xóa. Vui lòng kiểm tra lại.'); }); } function changeSyncKey(val) { syncKey = val.trim().replace(/[^a-zA-Z0-9]/g, ''); // npoint id chỉ gồm chữ và số document.getElementById('syncKeyInput').value = syncKey; localStorage.setItem('tienich_synckey', syncKey); if (syncKey) { loadFromCloud(); } else { setStatus('', 'Đã tắt đồng bộ đám mây.'); } } function updateCount() { document.getElementById('count').textContent = cards.length; document.getElementById('emptyState').style.display = cards.length ? 'none' : ''; } function addCard() { const colorIdx = cards.length % COLORS.length; const card = { id: Date.now(), emoji: EMOJIS[Math.floor(Math.random() * 8)], name: 'Tiện ích ' + (cards.length + 1), category: CATEGORIES[0], link: '', colorIdx }; cards.push(card); save(); renderCards(); setTimeout(() => { const el = document.getElementById('card-' + card.id); if (el) el.scrollIntoView({ behavior: 'smooth', block: 'center' }); }, 50); } function removeCard(id) { if(confirm('Bạn có chắc chắn muốn xóa tiện ích này?')) { cards = cards.filter(c => c.id !== id); save(); renderCards(); } } function updateField(id, field, val) { const c = cards.find(c => c.id === id); if (c) { c[field] = val; save(); } } function renderCards() { const grid = document.getElementById('grid'); const empty = document.getElementById('emptyState'); if(!grid) return; grid.innerHTML = ''; grid.appendChild(empty); updateCount(); cards.forEach(c => { const [a1, a2] = COLORS[c.colorIdx % COLORS.length]; const el = document.createElement('div'); el.className = 'card'; el.id = 'card-' + c.id; el.style.setProperty('--accent-color', a1); el.style.setProperty('--accent2-color', a2); el.innerHTML = `
${c.emoji}
${escHtml(c.name) || 'Chưa đặt tên'} ${escHtml(c.category)}
↗ Mở
`; grid.appendChild(el); }); } function handleOpen(e, id) { const c = cards.find(c => c.id === id); if (!c || !c.link) { e.preventDefault(); alert('Vui lòng nhập địa chỉ liên kết trước!'); return false; } e.currentTarget.href = c.link; return true; } function openPicker(e, id) { e.stopPropagation(); pickerTarget = id; const picker = document.getElementById('emojiPicker'); picker.innerHTML = EMOJIS.map(em => `` ).join(''); const rect = e.currentTarget.getBoundingClientRect(); picker.style.display = 'flex'; picker.style.top = (rect.bottom + window.scrollY + 6) + 'px'; picker.style.left = (rect.left + window.scrollX) + 'px'; } function selectEmoji(em) { if (!pickerTarget) return; updateField(pickerTarget, 'emoji', em); const card = document.getElementById('card-' + pickerTarget); if (card) card.querySelector('.card-emoji').textContent = em; document.getElementById('emojiPicker').style.display = 'none'; pickerTarget = null; } document.addEventListener('click', () => { const picker = document.getElementById('emojiPicker'); if(picker) picker.style.display = 'none'; pickerTarget = null; }); function escHtml(s) { return String(s||'').replace(/&/g,'&').replace(//g,'>').replace(/"/g,'"'); } function escAttr(s) { return String(s||'').replace(/"/g,'"'); } // Khởi động hệ thống renderCards(); if (syncKey) { const inputEl = document.getElementById('syncKeyInput'); if(inputEl) inputEl.value = syncKey; loadFromCloud(); } else { setStatus('', 'Đang lưu Offline. Bấm nút "🔄 Tải về" để tự động cấp mã đồng bộ Cloud mới.'); if (cards.length === 0) { const demos = [ { name: 'Google Drive', emoji: '📁', category: 'Tài liệu', link: 'https://drive.google.com' }, { name: 'Gmail', emoji: '📬', category: 'Công cụ', link: 'https://mail.google.com' }, { name: 'Google Calendar', emoji: '🗓', category: 'Quản lý', link: 'https://calendar.google.com' }, ]; demos.forEach((d, i) => { cards.push({ id: Date.now() + i, emoji: d.emoji, name: d.name, category: d.category, link: d.link, colorIdx: i }); }); localStorage.setItem('tienich_cards', JSON.stringify(cards)); renderCards(); } }
